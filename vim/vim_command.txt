# 連番数字を入力(マクロ使用)
i1<esc> 数字を一つ入れて、
qqyyp<c-a>q マクロを開始
マクロ実行 ( コマンド 100@q)

# 0始まりの数字を10進数として扱ってくれる設定
# 'nrformats' 'nf'	文字列	(既定では "octal,hex")
# vimではデフォで0始まりの数字は8進数と扱われ、<c-a>のインクリメントが
# 07でやると次は08でなく、10となってしまう。その際は以下の設定を行う。ｌｌ
set nf=""

# マクロ
# 記録
#ノーマルモードの状態で、「qとアルファベット1文字（マクロ名）」を入力すると
#記録が開始します。
qa

#記録終了
q

# 使用
# 10回マクロ名「a」を繰り返す
10@a

# 閲覧
:reg

# 削除
# 上書きするか
qa q
#以下入力で削除
:leg @a=''
